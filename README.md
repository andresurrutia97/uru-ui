## Install

```bash
npm install --save uru-ui
```

## Usage

```jsx
import { Button, InputNumber } from "uru-ui";

const App = () => {
  return(
    <>
      <Button color="primary">Click Me</Button>
      <InputNumber/>
    <>
  )
}
```
